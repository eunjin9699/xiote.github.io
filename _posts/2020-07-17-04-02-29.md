```
C-x o
  # (other-window) 
  
C-x 0
  # (delete-window)  
  
C-x 1
  # (delete-other-windows) 
  
  
C-x C-b
  # (list-buffers)
M-x delete-windows-on <RET> buffer <RET>
  # Delete windows showing the specified buffer. 
C-x b buffer <RET>
  # (switch-to-buffer) 
C-x 4 b buffer <RET>
  # (switch-to-buffer-other-window) 
C-x <LEFT>
  # (previous-buffer) 
C-x <RIGHT>
  # (next-buffer) 
```
