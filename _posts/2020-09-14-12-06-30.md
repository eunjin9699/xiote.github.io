```
# Command line 설치
echo "function gi() { curl -sLw "\n" https://www.toptal.com/developers/gitignore/api/\$@ ;}" >> \
~/.zshrc && source ~/.zshrc

# Command line 실행
gi vim,go >> .gitignore
```

```
# gitignore.io에서 자동생성
vim .gitignore

# staged 비우기
git rm -r --cached .

# 특정 파일만 빼기
git rm --cached package-lock.json

# .gitignore만 추가
git add .gitignore
git commmit


```
